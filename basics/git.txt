git init   - создаем git репозиторий из текущей папки
git remote add название ssh-ссылка   - сохраняем ssh-ссылку в переменную
git remote -v    - посмотреть все сохраненные ssh-ссылки
git add .   - просто пишем эту комманду каждый раз перед commit
git commit -m 'message'   - создаем новую версию
git push origin master   - закидываем версию в github
git pull origin master   - стягиваем новые изменения с github (если они есть в github, но их нет локально)
git clone ssh-ссылка    - копируем репозитроий с github (можно чужой)
git log   - посмотреть все коммиты

git branch  - показывает все ветки (* обозначена ветка на которой мы сейчас находимся)
git branch название   - создать новую ветку (просто создаем, но не переходим)
git branch -D ветка   - удалить ветку

git checkout ветка   - перейти на ветку
git checkout -b название   - создать ветку и перейти на нее


!!!!!!!!
git reset HEAD~  - отменяет последний коммит (как commit, но наоборот)
git restore .  - работает так же как add, но наоборот
